import { isIndexPage } from '../../../../utils';
import { groupBy } from 'lodash';
import { graphql as gatsbyGraphql, Link } from 'gatsby';

export const pageQuery = gatsbyGraphql`
  query verifyArticles {
    pages: allMdx(
      filter: {
        frontmatter: {
          samples: { eq: "verify" }
        }
      }
      sort: {frontmatter: {category: ASC}}
    ) {
      edges {
        node {
          __typename
          id
          fields {
            slug
          }
          frontmatter {
            title
            subtitle
            subcategory
          }
        }
      }
    }
  }
`;

<React.Fragment>
  {Object.entries(
    groupBy(
      props.data.pages.edges.map(edge => edge.node),
      'frontmatter.subcategory',
    ),
  ).map(([subcategory, pages]) => (
    <React.Fragment key={subcategory}>
      <h2>{subcategory}</h2>
      <ul>
        {pages
          .filter(node => !isIndexPage(node))
          .map(node => (
            <li key={node.id}>
              <Link to={`/${node.fields.slug}/`}>{node.frontmatter.title}</Link>
              <br />
              {node.frontmatter.subtitle}
            </li>
          ))}
      </ul>
    </React.Fragment>
  ))}
</React.Fragment>
