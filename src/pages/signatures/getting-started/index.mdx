import { isIndexPage } from '../../../utils';

import { graphql as gatsbyGraphql, Link } from 'gatsby';
export const pageQuery = gatsbyGraphql`
  query SignaturesGettingStarted {
    pages: allMdx(
      filter: {
        frontmatter: {
          product: { eq: "signatures" }
          category: { eq: "Getting Started" }
        }
      }
      sort: {frontmatter: {sort: ASC}}
    ) {
      edges {
        node {
          __typename
          id
          fields {
            slug
          }
          frontmatter {
            title
            subtitle
            category
          }
          internal {
            contentFilePath
          }
        }
      }
    }
  }
`;

Below you will find a list of topics to get you started with Criipto Signatures:

<ul>
  {props.data.pages.edges
    .map(edge => edge.node)
    .filter(node => !isIndexPage(node))
    .map(node => (
      <li key={node.id}>
        <Link to={`/${node.fields.slug}/`}>{node.frontmatter.title}</Link>
        <br />
        {node.frontmatter.subtitle}
      </li>
    ))}
</ul>
